import os
import os.path
import pandas as pd

#get a list of the files in the fruit folder given to us 
fruit_names = [x for x in os.listdir("fruit") if x.endswith("xlsx")]

#list of column names to be used later (dictated by the table above)
column_name=['type','food','form','price_per_lb', 'yield','lb_per_cup','price_per_cup']

#takes in the food_name and a file name and returns a data frame of 1 row and 6 column
def extract_fresh(food_name, fname):
    file_name = food_name+"/"+str(fname)
    panda_file = pd.read_excel(file_name)
    if panda_file.loc[2][0].find("Fresh") == 0:
        row = list()
        row.append(file_name.split('/')[0]) #gets the type
        row.append(fname.split('.')[0]) #gets the food
        row.append(panda_file.loc[2][0]) #gets the form
        row.append(panda_file.loc[2][1]) #gets the price_per_pounf
        row.append(panda_file.loc[2][3]) #gets the yield
        row.append(panda_file.loc[2][4]) #gets the lb_per_cup
        row.append(panda_file.loc[2][6]) #gets the price_per_cup
        row_df = pd.DataFrame([row], columns = column_name) #creates a df with the above info and the column names from above
        return row_df
            
fresh_fruit_df = [extract_fresh("fruit",x) for x in fruit_names] #returns a list of seperate data frames(one for each fruit file)
fresh_fruit_df = pd.concat(fresh_fruit_df)
