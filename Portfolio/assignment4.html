from fastcache import clru_cache
@clru_cache(maxsize=128,typed=False)

# query on USDA db, output json object (list of items)
def ndb_search(term):    
    # creating url with search term
    url = "https://api.nal.usda.gov/ndb/search/?format=json&q="+term
    
    # append remainder of url 
    #includes key and specifies the data source so it only grabs standard reference, not branded food products 
    search_url = url+"&sort=r&offset=0&api_key="+key+"&ds=Standard Reference"
    
    # use caching mechanism and retrieve the data
    request = requests.get(search_url)
    
    # make json object
    json_object = request.json()
    
    # return elements of json_obj (as "list", "item") when they are present or return an empty list when there is a KeyError
    try:
        list_out = json_object["list"]["item"]
    except KeyError:
        list_out=[]
    return(list_out)

def get_ndb_number(term):
    #gets the item list from the ndb_search function above
    item = ndb_search(term)
    #if the length of the item is 0 make it nan, otherwise return the ndbno in the first 
    if len(item) == 0:
        return(np.nan)
    else:
        return(item[0]['ndbno'])

def ndb_report(ndb_num):
    url = "https://api.nal.usda.gov/ndb/V2/reports"
    params = {"api_key" : "S8zY8kgwqX4fHLwCC9NR1H67JFOKSce11RI5i16U" 
              , "ndbno" : ndb_num}
    #retrieve data
    response = requests.get(url, params)
    #return response.json()
    json_obj = response.json()
    
    # return elements of json_obj when they are present and return an empty list when there is a KeyError
    try:
        list_out = json_obj["foods"][0]["food"]["nutrients"]
    except KeyError:
        list_out=[]
    return(list_out)
